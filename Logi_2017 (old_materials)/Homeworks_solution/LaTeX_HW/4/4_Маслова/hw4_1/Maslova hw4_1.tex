delete.dub<-function(k) {
  h<-c()
  if (length(k)!=0) {
    for (i in 1:(nrow(k)-1)) {
      for (j in (i+1):nrow(k)) {
        a<-sort(k[i,])
        b<-sort(k[j,])
        if ((all(a==b))&(!(j%in%h))) {
          h<-c(h,j)}}} 
    k<-k[-h,]} else {k<-"нет данной фигуры в графе"}
  return(k)}
