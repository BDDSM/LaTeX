Пример:
\begin{minted}[frame=lines,linenos]{R}
x<-c(1,4,-1,9,10,5)
n<-length(x)
max1<-x[1]
min1<-x[1]
nmax<-1
nmin<-1
for (i in 2:n ){
  if(x[i]>max1){max1<-x[i]
               nmax<-i}  
  else {if(x[i]<min1){min1<-x[i]
                     nmin<-i}}}
print(max1)
print(nmax)
print(min1)
print(nmin)
\end{minted}
