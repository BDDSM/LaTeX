minim<-function(gr) {
  n<-ncol(gr)
  min<-gr[1,2]
  ind<-c(1,2)
  for (i in 1:n) { for (j in 1:n) { 
    if (gr[i,j]<min) { 
      min<-gr[i,j]
      ind<-c(i,j)}}}
  return(c(min,ind))}

level<-function(g,v) {
  n<-ncol(g)
  if(v<=n) { 
    h<-g[v,] 
    for (ur in 2:(n-1)) {
      for (i in 1:n) { if(h[i]==(ur-1)) { for (j in 1:n) { 
        if((h[j]==0)&(g[i,j]==1)&(j!=v)) {h[j]<-(ur)}}}}}} else {print('error')} 
  return(h)} 


subgraph<- function (gr) { 
  A<-rep(0,ncol(gr))
  S<-list()
  b<-1
  for(i in 1:ncol(gr)){ 
    if(A[i]!=1) { 
      L<-level(gr,i)
      S[[b]]<-i
      for (j in 1:ncol(gr)){ 
        if (L[j]!=0) { 
          S[[b]]<-c(S[[b]],j)}}
      A[S[[b]]]<-1
      b<-b+1}}
  return(S)}

krus<-function(gr) { 
  g<-matrix(0,ncol=ncol(gr),nrow=ncol(gr))
  
  nelz<-matrix(0,ncol=ncol(gr),nrow=ncol(gr))
  k<-minim(gr)[1]
  a<-minim(gr)[2]
  b<-minim(gr)[3]
  
  s<-subgraph(g)
  
  while (length(s)>1) {
    if (nelz[a,b]==0) { 
      g[a,b]<-1
      g[b,a]<-1
      gr[a,b]<-+Inf
      gr[b,a]<-+Inf
      for (i in 1:ncol(gr)) {
        if(g[a,i]==1) { 
          nelz[b,i]<-1
          nelz[i,b]<-1}

          nelz[a,i]<-1
          nelz[i,a]<-1}}}
    k<-minim(gr)[1]
    a<-minim(gr)[2]
    b<-minim(gr)[3]
    s<-subgraph(g)}
  return(g)}
