\begin{minted}[breaklines,linenos]{R}
set.seed (23)
x<-sample (-100:100, 10, T, NULL)
print (x)
n<-length (x)
s<-0
for (j in 1:(n-1)) {
  s<-0 
  for (i in 1:(n-j)) {
    if (x[i]>x[i+1]) {
      t<-x[i]
      x[i]<-x[i+1]
      x[i+1]<-t
      s<-s+1
    }
  }
  if (s==0) {break} 
  else {}
}
f<-j
k<-f*n
print (x)
print (f)
print (k)
\end{minted}
