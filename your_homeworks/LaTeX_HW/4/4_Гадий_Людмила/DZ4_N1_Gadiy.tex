#Данный код находит корень уравнения на отрезке с заданной точностью методом бисекции
def bisec(f, a, b, e):
    an = a
    bn = b
    k = 0
    while abs(bn - an) / 2 > e:
        xn = (an + bn) / 2
        if f(an) * f(xn) <= 0:
            bn = xn
        else:
            an = xn
        k = k + 1
    return (an + bn)/2, k
\end{minted}
